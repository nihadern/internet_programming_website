SUPER_SERCET_WORD = "word"
word = list(SUPER_SERCET_WORD.lower())
current_guess = ["_" for i in range(len(word))]


def guessLetter(letter: str) -> bool:
    letter = letter.lower()
    try:
        letter_index = word.index(letter)
        print("Congratulations! You guessed a letter!")
        current_guess[letter_index] = word[letter_index]
        unguessed = current_guess.count("_")
        print("You have {} letters left".format(unguessed))
        print(''.join(current_guess))
        if (unguessed == 0):
            print("Congratulations! You have won the game!")
            return True
        else:
            return False
    except ValueError:
        print("Sorry! That guess is wrong!")
        return False


guessed = 0

while not guessed:
    guessed = guessLetter(input("Guess a letter: "))

import json
import requests
from datetime import datetime

with open("api.txt", 'r') as f:
    API_KEY = f.read().replace(" ", "").replace("\n", "")

API = 'https://api.openweathermap.org/data/2.5/weather?zip={},{}&appid={}'
data = requests.get(API.format(30144, 'us', API_KEY)).json()

print("Name:".ljust(25) + "{}".format(data["name"]))
temp = round(1.8*(data["main"]["temp"] - 273)+32, 2)
print("Current temprature:".ljust(25) + "{} degrees Fahrenheit".format(temp))
print("Atmospheric pressure:".ljust(25) + "{} hPa".format(data['main']['pressure']))
wind_speed = round(2.23694*data['wind']['speed'], 2)
print("Wind speed:".ljust(25) + "{} mph".format(wind_speed))
print("Wind direction:".ljust(25) + str(data["wind"]["deg"]))
report_time = datetime.fromtimestamp(data["dt"])
report_time = report_time.strftime("%Y-%m-%d %H:%M:%S")
print("Time of report:".ljust(25) + str(report_time))
